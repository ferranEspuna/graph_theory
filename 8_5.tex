    \noindent \textbf{Problem 4:}
    Let $T_{d, k}$ be a $(d-1)$-ary rooted tree of height $k$.

    \begin{enumerate}
        \item[(a)] Show that $\lambda_1(T_{d, k}) \leq 2 \sqrt{d - 1}$ \label{item:a}
        \item[(b)] Show that $\lim_{k \to \infty} \lambda_1(T_{d, k}) = 2 \sqrt{d - 1}$ \label{item:b}
    \end{enumerate}


    \paragraph{\textbf{Solution (by Ferran Espu√±a):}}
    Let $A$ be the adjacency matrix of $T_{d, k}$.
    For example, if $d = 3$ and $k = 3$, then

    \begin{center}

    \begin{align*}

        A = \begin{pmatrix}
            0  &  1  &  1  &  1  &  0  &  0 \\
            0  &  1  &  1  &  1  &  0  &  0
        \end{pmatrix}

    \end{align*}
    \end{center}
